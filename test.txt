class Bagel {
  string flavor;
  boolean toasted;

  Bagel Bagel(string f, boolean t) {
    this.flavor = f;
    this.toasted = t;
  }

  string describe() {
    string state = "raw";
    if (this.toasted) {
      state = "toasted";
    }
    return "A " + state + " " + this.flavor + " bagel.";
  }
}

Bagel myBreakfast = Bagel("everything", true);
print myBreakfast.describe();
